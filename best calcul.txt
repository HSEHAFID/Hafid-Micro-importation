        // حساب الضريبة
        function calculateTax() {
            let customsValue = parseFloat(customsValueInput.value);
            
            // إخفاء رسالة الخطأ
            valueError.style.display = 'none';
            customsValueInput.style.borderColor = '#ddd';
            
            // التحقق من صحة المدخلات
            if (isNaN(customsValue) || customsValue < 0) {
                valueError.textContent = 'يرجى إدخال قيمة جمركية صحيحة';
                valueError.style.display = 'block';
                customsValueInput.style.borderColor = 'var(--accent-color)';
                customsValueInput.focus();
                return;
            }
            
            if (customsValue > 100000000) {
                valueError.textContent = 'القيمة المدخلة كبيرة جداً، يرجى إدخال قيمة أصغر';
                valueError.style.display = 'block';
                customsValueInput.style.borderColor = 'var(--warning-color)';
                customsValueInput.focus();
                return;
            }
            
            // عرض مؤشر التحميل
            calculateBtn.classList.add('btn-loading');
            btnText.textContent = 'جاري الحساب...';
            
            // محاكاة تأخير للحساب
            setTimeout(() => {
                // حساب الحقوق الجمركية (5%)
                const customsDuty = customsValue * 0.05;
                
                // حساب الهامش الجزافي (30%)
                const provisionalMargin = (customsValue + customsDuty) * 0.3;
                
                // حساب الأساس الضريبي
                const taxBase = customsValue + customsDuty + provisionalMargin;
                
                // حساب الضريبة الجزافية الوحيدة (0.5%)
                const unifiedTax = taxBase * 0.005;
                
                // حساب الضريبة السنوية (24 عملية)
                const annualTax = unifiedTax * 24;
                
                // تحديث النتائج في الصفحة
                document.getElementById('inputValue').textContent = formatNumber(customsValue) + ' دج';
                document.getElementById('customsDuty').textContent = formatNumber(customsDuty) + ' دج';
                document.getElementById('provisionalMargin').textContent = formatNumber(provisionalMargin) + ' دج';
                document.getElementById('taxBase').textContent = formatNumber(taxBase) + ' دج';
                document.getElementById('unifiedTax').textContent = formatNumber(unifiedTax) + ' دج';
                document.getElementById('annualTax').textContent = formatNumber(annualTax) + ' دج';
                
                // إخفاء مؤشر التحميل
                calculateBtn.classList.remove('btn-loading');
                btnText.textContent = 'احسب الضريبة';
                
                // إضافة تأثير على النتائج
                animateResults();
            }, 800);
        }

        // تنسيق الأرقام
        function formatNumber(num) {
            return new Intl.NumberFormat('ar-DZ').format(num.toFixed(2));
        }

        // تأثيرات النتائج
        function animateResults() {
            const resultValues = document.querySelectorAll('.result-value');
            resultValues.forEach((value, index) => {
                value.style.opacity = '0';
                value.style.transform = 'translateY(10px)';
                
                setTimeout(() => {
                    value.style.transition = 'all 0.5s';
                    value.style.opacity = '1';
                    value.style.transform = 'translateY(0)';
                }, index * 100);
            });
        }
